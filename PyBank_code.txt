# # PyBank Data
import csv
import statistics

# #Set variables
tot_months= 0
tot_PL= 0.0
ch_PL = []
increase_prof = ["",0]
decrease_prof = ["",0]
prev_mon_PL= 0

# #Open the CSV File
with open("Resources/budget_data.csv", 'r') as csvfile:
    csvreader = csv.reader(csvfile, delimiter = ",")

    header = next(csvreader)

    for row in csvreader:
        # print(row)
        # Add Total number of months
        tot_months = tot_months + 1

        #Add profit/loss
        tot_PL = tot_PL + int(row[1])

        #Calculate Changes in PL then average
        if tot_months >1:
            ch_PL.append(int(row[1]) - prev_mon_PL)

        prev_mon_PL = int(row[1])

    print('Financial Analysis')
    print('--------------------')
    print(f'Total Months: {tot_months}')
    print(f'Total Profit/Loss: {tot_PL}')
    print(f'Average Change: {round(statistics.mean(ch_PL),2)}')
